    var Game_Winner = (S) => 
    {
        let N=S.length;
        A=0;
        B=0;
        
        for(i=0;i< N;i++){
        if(S[i]==='A')
        {
            A++;
        }
        if(S[i]==='D')
        {
            B++;
        }
        }
        if(A===B)
        {
        return 'Draw';
        }
        else
        {
        if(A>B)
        {
            return 'Aditya' ;
        }
        else
        {
            return 'Danish';
        }
        }
    }
